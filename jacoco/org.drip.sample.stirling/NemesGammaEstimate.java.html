public class NemesGammaEstimate
{

	public static final void main (
		final String[] argumentArray)
		throws Exception
	{
		EnvManager.InitEnv ("");

		int factorialCount = 12;

		NemesGamma nemesGamma = new NemesGamma (null);

		System.out.println ("\t|------------------------------------------------------------||");

		System.out.println ("\t|                  NEMES GAMMA APPROXIMATION                 ||");

		System.out.println ("\t|------------------------------------------------------------||");

		System.out.println ("\t|      L -> R:                                               ||");

		System.out.println ("\t|              - Gamma Index                                 ||");

		System.out.println ("\t|              - Gamma Value                                 ||");

		System.out.println ("\t|              - Nemes Estimate                              ||");

		System.out.println ("\t|------------------------------------------------------------||");

		for (int factorialIndex = 1; factorialIndex <= factorialCount; ++factorialIndex)
		{
			System.out.println (
				"\t| " + factorialIndex + " => " +
				NumberUtil.Factorial (factorialIndex - 1) + " |" +
				FormatUtil.FormatDouble (nemesGamma.evaluate (factorialIndex), 1, 0, 1.) + " ||"
			);
		}

		System.out.println ("\t|------------------------------------------------------------||");

		EnvManager.TerminateEnv();
	}
}
